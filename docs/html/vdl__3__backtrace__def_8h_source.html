27static inline void vdl_PrintBacktrace(const VDL_BACKTRACE_T bt)
28{
29 if (bt.FrameCount < 1)
30 return;
31 printf("Backtrace - %d stack frames:\n", bt.FrameCount);
32 char spaces[VDL_BACKTRACE_MAX_FRAME_NUM] = {'\0'};
33 int func_length[VDL_BACKTRACE_MAX_FRAME_NUM] = {0};
34 int max_func_length = -1;
35 vdl_for_i(bt.FrameCount)
36 {
37 func_length[i] = (int) strlen(bt.FunctionName[i]);
38 if (func_length[i] > max_func_length)
39 max_func_length = func_length[i];
40 }
41
42 for (int i = bt.FrameCount - 1; i >= 0; i--)
43 {
44 if (i != 0)
45 fputs(" ║═", stdout);
46 else
47 fputs(" ╚═", stdout);
48
49 int space_length = max_func_length - (int) strlen(bt.FunctionName[i]);
50
51 vdl_for_j(space_length) spaces[j] = ' ';
52 spaces[space_length] = '\0';
53
54 printf("[%d] Calling <%.*s>%s from %s:%d\n",
55 bt.FrameCount - i - 1,
56 (int) strlen(bt.FunctionName[i]),
57 bt.FunctionName[i],
58 spaces,
59 bt.FileName[i],
60 bt.Line[i]);
61 }
62}

84static inline void vdl_PushFrameToBacktrace(const VDL_FRAME_T fr)
85{
86 // Record the current frame (a call to the current function)
87 vdl_GlobalVar_Backtrace.Line[vdl_GlobalVar_Backtrace.FrameCount] = fr.Line;
88 vdl_GlobalVar_Backtrace.FunctionName[vdl_GlobalVar_Backtrace.FrameCount] = fr.FunctionName;
89 vdl_GlobalVar_Backtrace.FileName[vdl_GlobalVar_Backtrace.FrameCount] = fr.FileName;
90
91 vdl_GlobalVar_Backtrace.FrameCount++;
92
93 // Check if the stack depth exceed the limit.
94 vdl_Expect(vdl_GlobalVar_Backtrace.FrameCount <= VDL_BACKTRACE_MAX_FRAME_NUM,
95 VDL_EXCEPTION_EXCEED_STACK_LIMIT,
96 "Exceed the stack limit allowed by the backtrace [%d]!",
97 VDL_BACKTRACE_MAX_FRAME_NUM);
98}
