12static inline void vdl_SaveErrorMessage(const VDL_EXCEPTION_T exception_id,
13 const char *const function_name,
14 const char *const file_name,
15 const int line,
16 const char *const format,
17 ...)
18{
19 // If the format of the error message is empty, do not rewrite the buffer
20 if (format[0] == '\0')
21 return;
22
23 // Create the customized error message.
24 char message[VDL_ERROR_MESSAGE_MAX_LEN] = {0};
25 va_list ap;
26 va_start(ap, format);
27 int message_len = vsnprintf(message, VDL_ERROR_MESSAGE_MAX_LEN, format, ap);
28 va_end(ap);
29
30 // Check encoding error
31 if (message_len < 0)
32 {
33 puts("Encounter an encoding error! Buffer reset!");
34 message[0] = '\0';
35 }
36
37
38 // Rewrite the buffer
39 int len = snprintf(vdl_GlobalVar_ErrorMessage,
40 VDL_ERROR_MESSAGE_MAX_LEN,
41 "[E%03d] Error raised by <%s> at %s:%d: %s\n",
42 exception_id,
43 function_name,
44 file_name,
45 line,
46 message);
47
48 // Check encoding error
49 if (len < 0)
50 {
51 puts("Encounter an encoding error! Buffer reset!");
52 vdl_GlobalVar_ErrorMessage[0] = '\0';
53 }
54}

60static inline void vdl_ExceptionRegisterCleanUp(void *const object, void (*const clean_up_function)(void *))
61{
62 int current_len = vdl_GlobalVar_ExceptionCleanUp.Length;
63 // Store the object and the cleanup function
64 vdl_GlobalVar_ExceptionCleanUp.Object[current_len] = object;
65 vdl_GlobalVar_ExceptionCleanUp.CleanUpFunction[current_len] = clean_up_function;
66 vdl_GlobalVar_ExceptionCleanUp.Length++;
67
68 // If the cleanup stack is full, throw an exception
69 if (current_len >= VDL_EXCEPTION_CLEAN_UP_MAX_FUNCTION_NUM)
70 {
71 vdl_Throw(VDL_EXCEPTION_EXCEED_CLEANUP_LIMIT,
72 "Exceed the cleanup limit allowed by the exception [%d]!",
73 VDL_EXCEPTION_CLEAN_UP_MAX_FUNCTION_NUM);
74 }
75}

77static inline void vdl_ExceptionCleanUp(void)
78{
79 for (int i = vdl_GlobalVar_ExceptionCleanUp.Length - 1; i >= 0; i--)
80 {
81 vdl_GlobalVar_ExceptionCleanUp.CleanUpFunction[i](vdl_GlobalVar_ExceptionCleanUp.Object[i]);
82 }
83 vdl_GlobalVar_ExceptionCleanUp.Length = 0;
84}
