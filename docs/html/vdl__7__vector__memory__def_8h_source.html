8static inline VDL_VECTOR_P vdl_vector_primitive_NewEmpty_BT(const VDL_TYPE_T type, const int capacity)
9{
10 vdl_CheckRequestedCapacity(capacity);
11
12 // Create a vector and copy in the content.
13 VDL_VECTOR_P v = vdl_Malloc(sizeof(VDL_VECTOR_T), 1);
14 VDL_VECTOR_P local_v = &(VDL_VECTOR_T){.Capacity = capacity,
15 .Mode = VDL_MODE_HEAP,
16 .Type = type,
17 .Class = VDL_CLASS_VECTOR,
18 .Length = 0,
19 .Attribute = NULL,
20 .Data = NULL};
21 memcpy(v, local_v, sizeof(VDL_VECTOR_T));
22
23 // Allocate memory for the data container.
24 v->Data = vdl_Malloc((size_t) capacity * VDL_TYPE_SIZE[type], 1);
25
26 // Record the vector in the vector table.
27 vdl_GarbageCollectorRecord(v);
28
29 vdl_ExceptionDeregisterCleanUp(v);
30 vdl_ExceptionDeregisterCleanUp(v->Data);
31 return v;
32}

92static inline VDL_VECTOR_P vdl_vector_primitive_NewByVariadic_BT(const VDL_TYPE_T type, const int length, ...)
93{
94 VDL_VECTOR_P v = vdl_vector_primitive_NewEmpty(type, length);
95 v->Length = length;
96
97 va_list ap;
98 va_start(ap, length);
99
100 switch (type)
101 {
102 case VDL_TYPE_CHAR:
103 {
104 vdl_for_i(length) vdl_vector_primitive_SetChar(v, i, (char) va_arg(ap, int));
105 break;
106 }
107 case VDL_TYPE_INT:
108 {
109 vdl_for_i(length) vdl_vector_primitive_SetInt(v, i, va_arg(ap, int));
110 break;
111 }
112 case VDL_TYPE_DOUBLE:
113 {
114 vdl_for_i(length) vdl_vector_primitive_SetDouble(v, i, va_arg(ap, double));
115 break;
116 }
117 case VDL_TYPE_VECTOR_POINTER:
118 {
119 vdl_for_i(length) vdl_vector_primitive_SetVectorPointer(v, i, va_arg(ap, VDL_VECTOR_P));
120 break;
121 }
122 }
123
124 va_end(ap);
125 return v;
126}

128static inline void vdl_vector_primitive_Reserve_BT(VDL_VECTOR_T *const v, const int capacity)
129{
130 vdl_CheckNullVectorAndNullContainer(v);
131 vdl_CheckMode(v->Mode, VDL_MODE_HEAP);
132 vdl_CheckRequestedCapacity(capacity);
133
134 if (v->Capacity >= capacity)
135 return;
136
137 size_t target_capacity = (size_t) v->Capacity;
138
139 // Allocate enough space
140 // Switch to arithmetic growth policy after 500KB
141 while (target_capacity < (size_t) capacity)
142 {
143 static const size_t MEM_500KB = 500 * 1024;
144 if (target_capacity * VDL_TYPE_SIZE[v->Type] < MEM_500KB)
145 target_capacity = target_capacity * 2 + 8;
146 else
147 target_capacity += MEM_500KB / VDL_TYPE_SIZE[v->Type];
148 }
149 if (target_capacity > VDL_VECTOR_MAX_CAPACITY)
150 target_capacity = VDL_VECTOR_MAX_CAPACITY;
151
152 void *buffer = vdl_Malloc(target_capacity * VDL_TYPE_SIZE[v->Type], 1);
153 v->Data = buffer;
154 v->Capacity = (int) target_capacity;
155
156 vdl_ExceptionDeregisterCleanUp(buffer);
157}
